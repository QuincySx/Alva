// INPUT:  std::any::{Any, TypeId}, std::collections::HashMap, std::sync::Arc, parking_lot::RwLock
// OUTPUT: Caps
// POS:    Capability registry — typed service locator using TypeId keys.

use std::any::{Any, TypeId};
use std::collections::HashMap;
use std::sync::Arc;
use parking_lot::RwLock;

/// Thread-safe capability registry.
///
/// Stores `Arc<T>` values keyed by `TypeId`. Each layer registers its
/// capabilities at init time; other layers discover them at runtime.
#[derive(Clone)]
pub struct Caps {
    slots: Arc<RwLock<HashMap<TypeId, Box<dyn Any + Send + Sync>>>>,
}

impl Caps {
    pub fn new() -> Self {
        Self {
            slots: Arc::new(RwLock::new(HashMap::new())),
        }
    }

    /// Register a capability. Overwrites any previous value of the same type.
    pub fn provide<T: Send + Sync + 'static>(&self, val: Arc<T>) {
        let mut slots = self.slots.write();
        slots.insert(TypeId::of::<T>(), Box::new(val));
    }

    /// Look up a capability by type. Returns None if not registered.
    pub fn get<T: Send + Sync + 'static>(&self) -> Option<Arc<T>> {
        let slots = self.slots.read();
        slots
            .get(&TypeId::of::<T>())
            .and_then(|b| b.downcast_ref::<Arc<T>>())
            .cloned()
    }

    /// Look up a capability, panic if not registered.
    ///
    /// Use this for capabilities that MUST exist (registered during init).
    /// The panic message includes the type name for easy debugging.
    pub fn require<T: Send + Sync + 'static>(&self) -> Arc<T> {
        self.get::<T>().unwrap_or_else(|| {
            panic!(
                "bus: required capability not registered: {}",
                std::any::type_name::<T>()
            )
        })
    }

    /// Check if a capability is registered.
    pub fn has<T: Send + Sync + 'static>(&self) -> bool {
        let slots = self.slots.read();
        slots.contains_key(&TypeId::of::<T>())
    }
}

impl Default for Caps {
    fn default() -> Self {
        Self::new()
    }
}

// INPUT:  std::any::TypeId, std::collections::HashMap, std::sync::Arc,
//         tokio::sync::broadcast, parking_lot::Mutex
// OUTPUT: BusEvent (trait), EventBus
// POS:    Typed event bus — each event type gets its own broadcast channel.

use std::any::TypeId;
use std::collections::HashMap;
use std::sync::Arc;
use parking_lot::Mutex;
use tokio::sync::broadcast;

/// Marker trait for bus events. Events must be Clone + Send + Sync.
pub trait BusEvent: Clone + Send + Sync + 'static {}

/// Default channel capacity per event type.
const DEFAULT_CHANNEL_CAPACITY: usize = 64;

/// Typed event bus backed by per-type broadcast channels.
///
/// - `emit()` is non-blocking (uses `send`, drops if no subscribers).
/// - `subscribe()` returns a `broadcast::Receiver` for async consumption.
/// - Channels are created lazily on first `emit` or `subscribe`.
#[derive(Clone)]
pub struct EventBus {
    channels: Arc<Mutex<HashMap<TypeId, Box<dyn std::any::Any + Send + Sync>>>>,
}

impl EventBus {
    pub fn new() -> Self {
        Self {
            channels: Arc::new(Mutex::new(HashMap::new())),
        }
    }

    /// Subscribe to events of type E. Returns a broadcast receiver.
    pub fn subscribe<E: BusEvent>(&self) -> broadcast::Receiver<E> {
        let mut channels = self.channels.lock();
        let sender = channels
            .entry(TypeId::of::<E>())
            .or_insert_with(|| {
                let (tx, _) = broadcast::channel::<E>(DEFAULT_CHANNEL_CAPACITY);
                Box::new(tx)
            })
            .downcast_ref::<broadcast::Sender<E>>()
            .expect("bus: event type mismatch in channel map");
        sender.subscribe()
    }

    /// Emit an event. Non-blocking. No-op if no subscribers.
    pub fn emit<E: BusEvent>(&self, event: E) {
        let mut channels = self.channels.lock();
        if let Some(boxed) = channels.get(&TypeId::of::<E>()) {
            if let Some(tx) = boxed.downcast_ref::<broadcast::Sender<E>>() {
                let _ = tx.send(event);
            }
        } else {
            // Create the channel so future subscribers see it, but drop this event.
            let (tx, _) = broadcast::channel::<E>(DEFAULT_CHANNEL_CAPACITY);
            let _ = tx.send(event);
            channels.insert(TypeId::of::<E>(), Box::new(tx));
        }
    }

    /// Check if there are active subscribers for an event type.
    pub fn has_subscribers<E: BusEvent>(&self) -> bool {
        let channels = self.channels.lock();
        channels
            .get(&TypeId::of::<E>())
            .and_then(|b| b.downcast_ref::<broadcast::Sender<E>>())
            .map(|tx| tx.receiver_count() > 0)
            .unwrap_or(false)
    }
}

impl Default for EventBus {
    fn default() -> Self {
        Self::new()
    }
}

// INPUT:  std::sync::Arc, tokio::sync::broadcast, parking_lot::RwLock
// OUTPUT: StateCell<T>
// POS:    Observable state cell — write triggers broadcast notification to watchers.

use std::sync::Arc;
use parking_lot::RwLock;
use tokio::sync::broadcast;

const WATCH_CHANNEL_CAPACITY: usize = 16;

/// A shared, observable value.
///
/// Reads are lock-free (RwLock read guard). Writes broadcast the new
/// value to all watchers. Suitable for cross-layer state like
/// "current permission mode" or "token usage ratio".
pub struct StateCell<T: Clone + Send + Sync + 'static> {
    value: Arc<RwLock<T>>,
    notify: broadcast::Sender<T>,
}

impl<T: Clone + Send + Sync + 'static> StateCell<T> {
    /// Create a new cell with an initial value.
    pub fn new(initial: T) -> Self {
        let (tx, _) = broadcast::channel(WATCH_CHANNEL_CAPACITY);
        Self {
            value: Arc::new(RwLock::new(initial)),
            notify: tx,
        }
    }

    /// Read the current value.
    pub fn get(&self) -> T {
        self.value.read().clone()
    }

    /// Write a new value and notify all watchers.
    pub fn set(&self, val: T) {
        *self.value.write() = val.clone();
        let _ = self.notify.send(val);
    }

    /// Modify the value in place and notify watchers with the new value.
    pub fn update(&self, f: impl FnOnce(&mut T)) {
        let mut guard = self.value.write();
        f(&mut guard);
        let _ = self.notify.send(guard.clone());
    }

    /// Subscribe to value changes. Returns a broadcast receiver.
    pub fn watch(&self) -> broadcast::Receiver<T> {
        self.notify.subscribe()
    }
}

impl<T: Clone + Send + Sync + 'static> Clone for StateCell<T> {
    fn clone(&self) -> Self {
        Self {
            value: self.value.clone(),
            notify: self.notify.clone(),
        }
    }
}

#[cfg(test)]
mod tests {

    #[test]

// INPUT:  Caps, EventBus, BusEvent
// OUTPUT: BusHandle
// POS:    Clone-friendly facade distributed to all layers.

use std::sync::Arc;
use tokio::sync::broadcast;

use crate::caps::Caps;
use crate::event::{BusEvent, EventBus};

/// Clone-friendly handle distributed to each layer.
///
/// Provides the full bus API: capability registration/discovery,
/// event emit/subscribe. Cloning shares the underlying state.
#[derive(Clone)]
pub struct BusHandle {
    pub(crate) caps: Caps,
    pub(crate) events: EventBus,
}

impl BusHandle {
    // -- Caps delegation --

    /// Register a capability.
    pub fn provide<T: Send + Sync + 'static>(&self, val: Arc<T>) {
        self.caps.provide(val);
    }

    /// Look up a capability. Returns None if not registered.
    pub fn get<T: Send + Sync + 'static>(&self) -> Option<Arc<T>> {
        self.caps.get()
    }

    /// Look up a capability, panic if not registered.
    pub fn require<T: Send + Sync + 'static>(&self) -> Arc<T> {
        self.caps.require()
    }

    /// Check if a capability is registered.
    pub fn has<T: Send + Sync + 'static>(&self) -> bool {
        self.caps.has::<T>()
    }

    // -- Event delegation --

    /// Emit an event (non-blocking).
    pub fn emit<E: BusEvent>(&self, event: E) {
        self.events.emit(event);
    }

    /// Subscribe to events of type E.
    pub fn subscribe<E: BusEvent>(&self) -> broadcast::Receiver<E> {
        self.events.subscribe()
    }
}

impl std::fmt::Debug for BusHandle {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        f.debug_struct("BusHandle").finish_non_exhaustive()
    }
}

use crate::caps::Caps;
use crate::event::EventBus;
use crate::handle::BusHandle;

/// The bus owner. Created once at application startup.
///
/// Call `handle()` to produce `BusHandle` instances for each layer.
/// The `Bus` itself is not Clone — it represents unique ownership.
/// All `BusHandle`s share the same underlying Caps + EventBus.
pub struct Bus {

    events: EventBus,

    pub fn new() -> Self {
        Self {
            caps: Caps::new(),
            events: EventBus::new(),
        }
    }

    /// Produce a handle for distribution to a layer.
    pub fn handle(&self) -> BusHandle {
        BusHandle {
            caps: self.caps.clone(),
            events: self.events.clone(),
        }
    }
}

impl Default for Bus {
    fn default() -> Self {
        Self::new()
    }
}

//! Integration test — verifies the full bus round-trip across simulated layers.

use alva_agent_bus::{Bus, BusEvent, BusHandle};
use std::sync::Arc;

// -- Shared event definitions (would live in alva-types) --

#[derive(Clone, Debug, PartialEq)]
struct ToolExecuted {
    tool_name: String,
    session_id: String,
}
impl BusEvent for ToolExecuted {}

#[derive(Clone, Debug, PartialEq)]
struct TokenUsageUpdated {
    used: usize,
    total: usize,
}
impl BusEvent for TokenUsageUpdated {}

// -- Shared capability trait (would live in alva-types) --

trait TokenAccounting: Send + Sync {
    fn current_usage(&self) -> (usize, usize);
}

struct MockTokenAccounting;
impl TokenAccounting for MockTokenAccounting {
    fn current_usage(&self) -> (usize, usize) {
        (3000, 4096)
    }
}

// -- Simulated layers --

fn init_engine_layer(bus: &BusHandle) {
    // Engine registers token accounting capability
    bus.provide::<dyn TokenAccounting>(Arc::new(MockTokenAccounting));
}

fn init_tool_layer(bus: &BusHandle) {
    // After tool execution, emit event
    bus.emit(ToolExecuted {
        tool_name: "shell".into(),
        session_id: "sess-1".into(),
    });
}

async fn context_layer_subscribe(bus: &BusHandle) -> ToolExecuted {
    let mut rx = bus.subscribe::<ToolExecuted>();
    rx.recv().await.unwrap()
}

#[tokio::test]
async fn full_bus_round_trip() {
    let bus = Bus::new();

    // Distribute handles
    let engine_handle = bus.handle();
    let tool_handle = bus.handle();
    let context_handle = bus.handle();

    // Init layers
    init_engine_layer(&engine_handle);

    // Context layer subscribes BEFORE tool layer emits
    let context_task = {
        let h = context_handle.clone();
        tokio::spawn(async move { context_layer_subscribe(&h).await })
    };

    // Small delay to ensure subscription is active
    tokio::task::yield_now().await;

    // Tool layer emits
    init_tool_layer(&tool_handle);

    // Context layer receives
    let received = context_task.await.unwrap();
    assert_eq!(received.tool_name, "shell");
    assert_eq!(received.session_id, "sess-1");

    // Any handle can discover engine's capability
    let accounting = context_handle.require::<dyn TokenAccounting>();
    assert_eq!(accounting.current_usage(), (3000, 4096));
}

#[tokio::test]
async fn state_cell_cross_handle() {
    use alva_agent_bus::StateCell;

    let bus = Bus::new();
    let h1 = bus.handle();
    let h2 = bus.handle();

    // Share a StateCell via capability
    let cell = StateCell::new(0u32);
    h1.provide(Arc::new(cell.clone()));

    // Another handle reads and watches
    let cell_ref = h2.require::<StateCell<u32>>();
    let mut rx = cell_ref.watch();

    cell.set(42);
    assert_eq!(cell_ref.get(), 42);
    assert_eq!(rx.recv().await.unwrap(), 42);
}

#[test]
fn concurrent_provide_and_get() {
    use std::thread;

    let bus = Bus::new();
    let handles: Vec<_> = (0..10).map(|_| bus.handle()).collect();

    // 10 threads each provide a different concrete type
    let threads: Vec<_> = handles
        .into_iter()
        .enumerate()
        .map(|(i, h)| {
            thread::spawn(move || {
                // Each thread provides a unique type via a wrapper
                #[derive(Debug)]
                struct Val(usize);
                h.provide(Arc::new(Val(i)));
            })
        })
        .collect();

    for t in threads {
        t.join().unwrap();
    }
}
